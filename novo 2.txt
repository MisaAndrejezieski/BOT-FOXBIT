"""
BOT DE TRADING AUTOMÁTICO PARA FOXBIT
Estratégia: COMPRA quando cair -2% | VENDA quando subir +4%
"""

import time
import hmac
import hashlib
import json
from datetime import datetime, date
import requests
from config import Config

class FoxbitBot:
    def __init__(self):
        self.config = Config()
        self.session = requests.Session()
        self.capital = 0
        self.position = None
        self.price_history = []
        self.trades_today = 0
        self.last_trade_date = date.today()
        self.total_trades = []
        
        print("\n" + "="*60)
        print("🤖 BOT FOXBIT - ESTRATÉGIA -2%/+4%")
        print("="*60)
        print(f"📊 Par: {self.config.SYMBOL.upper()}")
        print(f"📈 Compra: {self.config.BUY_THRESHOLD}% | Venda: {self.config.SELL_THRESHOLD}%")
        print(f"🛑 Stop Loss: {self.config.STOP_LOSS}%")
        print("="*60)
        
        self.check_credentials()
        self.update_balance()
    
    def check_credentials(self):
        """Verifica se as credenciais estão configuradas"""
        if not self.config.API_KEY or not self.config.API_SECRET:
            print("\n❌ ERRO: API Key ou Secret não configuradas!")
            print("📝 Configure no arquivo .env ou gere em:")
            print("   https://app.foxbit.com.br/profile/api-key")
            exit(1)
    
    def sign_request(self, method, path, params=None, body=None):
        """Gera assinatura para autenticação na Foxbit [citation:4]"""
        timestamp = str(int(time.time() * 1000))
        
        # Monta query string
        query_string = ""
        if params:
            query_string = "&".join([f"{k}={v}" for k, v in sorted(params.items())])
        
        # Monta body string
        body_string = ""
        if body:
            body_string = json.dumps(body, separators=(',', ':'))
        
        # Cria string para assinar
        text_to_sign = f"{timestamp}{method}{path}{query_string}{body_string}"
        
        # Gera assinatura HMAC-SHA256
        signature = hmac.new(
            self.config.API_SECRET.encode('utf-8'),
            text_to_sign.encode('utf-8'),
            hashlib.sha256
        ).hexdigest()
        
        headers = {
            'X-FB-ACCESS-KEY': self.config.API_KEY,
            'X-FB-ACCESS-SIGNATURE': signature,
            'X-FB-ACCESS-TIMESTAMP': timestamp,
            'Content-Type': 'application/json'
        }
        
        return headers
    
    def api_request(self, method, endpoint, params=None, data=None):
        """Faz requisição autenticada à API da Foxbit"""
        url = f"{self.config.BASE_URL}{endpoint}"
        headers = self.sign_request(method, endpoint, params, data)
        
        try:
            if method == 'GET':
                response = self.session.get(url, headers=headers, params=params)
            elif method == 'POST':
                response = self.session.post(url, headers=headers, json=data)
            
            if response.status_code == 200:
                return response.json()
            else:
                print(f"❌ Erro API: {response.status_code} - {response.text}")
                return None
        except Exception as e:
            print(f"❌ Erro requisição: {e}")
            return None
    
    def get_ticker(self):
        """Busca preço atual do Bitcoin [citation:1]"""
        result = self.api_request('GET', '/ticker', params={'market_symbol': self.config.SYMBOL})
        if result and 'data' in result:
            return float(result['data'][0]['last'])
        return None
    
    def get_balance(self):
        """Consulta saldo da conta [citation:4]"""
        result = self.api_request('GET', '/accounts/balance')
        if result and 'data' in result:
            balances = {}
            for item in result['data']:
                balances[item['currency_symbol']] = {
                    'balance': float(item['balance']),
                    'available': float(item['available'])
                }
            return balances
        return {}
    
    def update_balance(self):
        """Atualiza saldo atual"""
        balances = self.get_balance()
        if balances:
            brl = balances.get('brl', {}).get('available', 0)
            btc = balances.get('btc', {}).get('available', 0)
            self.capital = brl + (btc * self.get_ticker() if self.get_ticker() else 0)
            print(f"\n💰 Saldo: R$ {self.capital:.2f} | BTC: {btc:.6f}")
    
    def create_order(self, side, quantity):
        """
        Cria ordem de compra ou venda
        side: 'BUY' ou 'SELL'
        quantity: quantidade em BTC
        """
        order_data = {
            'market_symbol': self.config.SYMBOL,
            'side': side,
            'type': 'MARKET',
            'quantity': str(quantity)
        }
        
        result = self.api_request('POST', '/orders', data=order_data)
        if result and 'data' in result:
            return result['data']
        return None
    
    def should_buy(self, current_price):
        """Analisa se deve comprar"""
        if len(self.price_history) < 2:
            return False
        
        # Calcula queda percentual
        last_price = self.price_history[-2]
        drop = (current_price - last_price) / last_price * 100
        
        if drop <= self.config.BUY_THRESHOLD:
            print(f"\n📉 Queda detectada: {drop:.2f}%")
            return True
        return False
    
    def should_sell(self, current_price):
        """Analisa se deve vender"""
        if not self.position:
            return False
        
        buy_price = self.position['price']
        profit = (current_price - buy_price) / buy_price * 100
        
        if profit >= self.config.SELL_THRESHOLD:
            print(f"\n📈 Lucro atingido: {profit:.2f}%")
            return True
        elif profit <= self.config.STOP_LOSS:
            print(f"\n⚠️ Stop loss acionado: {profit:.2f}%")
            return True
        return False
    
    def reset_daily_counter(self):
        """Reseta contador de trades diário"""
        today = date.today()
        if today > self.last_trade_date:
            self.trades_today = 0
            self.last_trade_date = today
    
    def execute_buy(self, price):
        """Executa compra"""
        if self.trades_today >= self.config.MAX_TRADES_PER_DAY:
            print(f"⚠️ Limite diário de {self.config.MAX_TRADES_PER_DAY} trades atingido")
            return False
        
        # Pega saldo BRL disponível
        balances = self.get_balance()
        brl_available = balances.get('brl', {}).get('available', 0)
        
        if brl_available < 10:  # Mínimo R$10
            print("❌ Saldo BRL insuficiente")
            return False
        
        # Calcula quantidade baseada no saldo
        quantity = brl_available / price
        # Arredonda para 3 casas decimais (mínimo Foxbit)
        quantity = round(quantity, 3)
        
        if quantity < 0.001:
            print("❌ Quantidade mínima não atingida (0.001 BTC)")
            return False
        
        print(f"\n🟢 COMPRANDO {quantity:.6f} BTC a R$ {price:,.2f}")
        
        order = self.create_order('BUY', quantity)
        if order:
            self.position = {
                'price': price,
                'quantity': quantity,
                'time': datetime.now()
            }
            self.trades_today += 1
            print(f"✅ Compra executada! ID: {order.get('id', 'N/A')}")
            
            # Registra trade
            self.total_trades.append({
                'type': 'BUY',
                'price': price,
                'quantity': quantity,
                'time': datetime.now()
            })
            return True
        return False
    
    def execute_sell(self, price):
        """Executa venda"""
        if not self.position:
            return False
        
        quantity = self.position['quantity']
        
        print(f"\n🔴 VENDENDO {quantity:.6f} BTC a R$ {price:,.2f}")
        
        order = self.create_order('SELL', quantity)
        if order:
            buy_price = self.position['price']
            profit = (price - buy_price) / buy_price * 100
            profit_abs = (price - buy_price) * quantity
            
            print(f"✅ Venda executada! Lucro: {profit:.2f}% (R$ {profit_abs:.2f})")
            
            # Registra trade
            self.total_trades.append({
                'type': 'SELL',
                'price': price,
                'quantity': quantity,
                'profit': profit,
                'profit_abs': profit_abs,
                'time': datetime.now()
            })
            
            self.position = None
            return True
        return False
    
    def run(self, interval=60):
        """
        Loop principal do bot
        interval: segundos entre cada verificação
        """
        print(f"\n🔍 Monitorando mercado... (verificando a cada {interval}s)")
        print("="*60)
        
        try:
            while True:
                # Reseta contador diário
                self.reset_daily_counter()
                
                # Busca preço atual
                price = self.get_ticker()
                if not price:
                    print("❌ Erro ao buscar preço")
                    time.sleep(interval)
                    continue
                
                # Atualiza histórico
                self.price_history.append(price)
                if len(self.price_history) > 100:
                    self.price_history.pop(0)
                
                # Mostra status
                print(f"\n⏰ {datetime.now().strftime('%H:%M:%S')}")
                print(f"💰 BTC: R$ {price:,.2f}")
                
                if self.position:
                    profit = (price - self.position['price']) / self.position['price'] * 100
                    print(f"📊 Posição: {self.position['quantity']:.6f} BTC")
                    print(f"📈 Lucro: {profit:.2f}%")
                    print(f"🎯 Alvo: +{self.config.SELL_THRESHOLD}% | Stop: {self.config.STOP_LOSS}%")
                else:
                    print(f"💤 Aguardando queda de {self.config.BUY_THRESHOLD}%")
                
                print(f"📊 Trades hoje: {self.trades_today}/{self.config.MAX_TRADES_PER_DAY}")
                
                # Lógica de trading
                if not self.position:
                    if self.should_buy(price):
                        self.execute_buy(price)
                else:
                    if self.should_sell(price):
                        self.execute_sell(price)
                        self.update_balance()
                
                time.sleep(interval)
                
        except KeyboardInterrupt:
            self.shutdown()
    
    def shutdown(self):
        """Desliga o bot com segurança"""
        print("\n\n🛑 Desligando bot...")
        
        # Se tiver posição aberta, pergunta se quer vender
        if self.position:
            print("\n⚠️ Posição aberta detectada!")
            resp = input("Deseja vender agora? (s/N): ")
            if resp.lower() == 's':
                price = self.get_ticker()
                if price:
                    self.execute_sell(price)
        
        self.show_report()
    
    def show_report(self):
        """Mostra relatório de operações"""
        print("\n" + "="*60)
        print("📊 RELATÓRIO DE OPERAÇÕES")
        print("="*60)
        
        buys = [t for t in self.total_trades if t['type'] == 'BUY']
        sells = [t for t in self.total_trades if t['type'] == 'SELL']
        
        print(f"\n📈 Total de trades: {len(sells)}")
        
        if sells:
            profits = [t['profit'] for t in sells]
            profits_abs = [t['profit_abs'] for t in sells]
            
            print(f"   Trades com lucro: {len([p for p in profits if p > 0])}")
            print(f"   Trades com prejuízo: {len([p for p in profits if p < 0])}")
            print(f"   Lucro médio: {sum(profits)/len(profits):.2f}%")
            print(f"   Maior lucro: {max(profits):.2f}%")
            print(f"   Maior prejuízo: {min(profits):.2f}%")
            print(f"   Lucro total: R$ {sum(profits_abs):.2f}")
        
        # Atualiza saldo final
        self.update_balance()
        print(f"\n💰 Saldo final: R$ {self.capital:.2f}")
        print("="*60)
